\begin{exec}
LEt $f$ be a continuous real-valued function such that
\[
\int_0^1 x^n f(x)dx=0 \text{ for } n=0,1,2,\cdots
\]
Show that $f$ is the zero function.
\end{exec}
\proof
\qed

\begin{exec}
Let $C[0,1]$ denots the set of real-valued continuous functions on 
$[0,1]$. Show that there is a unique $f\in C[0,1]$ such that 
\[
f(x)=\int_0^{x/2} f(t)dt
\]
for all $x\in [0,1]$.
\end{exec}
